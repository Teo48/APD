- Se cloneaza repo-ul de lucru
git clone nume-git
cd git

- Se schimba lucrul pe branch
# Parametrul -b creeaza branch-ul inainte sa se faca switch pe el
git checkout -b nume_branch

- Se lucreaza in folderul aferent
# De exemplu teme/tema3
# Se adauga fisierele necesare
git add fisiere
# Se creeaza un commit
git commit -m "mesaj"
# Se poate folosi si cu parametrul -s
# Parametrul -s adauga o linie "Signed-off-by pentru a semna commit-ul cu numele si mail-ul
- Dupa ce s-au creat toate commit-urile, faci push la branch pe repo
# Verifici ca nu mai sunt fisiere nemodificate/neadaugate
git status
# Output asteptat
On branch nume_branch
nothing to commit, working tree clean

# Verifici commit-urile, mesajele etc.
git log --pretty --graph

# Faci push la commit-uri
git push origin nume_branch
# !!! Nu faci git push pe master/main
# Primesti un link de github in terminal, il deschizi si creezi un Pull Request
